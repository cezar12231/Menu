-- Vari√°vel de controle
local rodando = false

-- Fun√ß√£o do Auto Farm
local function autoFarm()
    while rodando do
        local valor = tonumber(TextBox.Text)
        if valor and valor > 0 then
            local negativo = "-" .. tostring(math.abs(valor))
            local args = {
                [1] = negativo
            }
            game:GetService("ReplicatedStorage").CratesUtilities.Remotes.RemoveMoney:FireServer(unpack(args))
        end
        task.wait(0.05) -- muito r√°pido (20x por segundo)
    end
end

-- Ativa o auto farm
StartButton.MouseButton1Click:Connect(function()
    if not rodando then
        rodando = true
        task.spawn(autoFarm)
    end
end)

-- Para o auto farm
StopButton.MouseButton1Click:Connect(function()
    rodando = false
end)
